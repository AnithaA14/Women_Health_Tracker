        function login() {
    var username = document.getElementById("username").value.trim();
    var password = document.getElementById("password").value.trim();
    var loginMessage = document.getElementById("loginMessage");

    if (username !== "" && password !== "") { // Allows any username and password
        loginMessage.style.display = "block";
        loginMessage.textContent = "Login successful! Redirecting...";
        loginMessage.style.color = "green";

        setTimeout(() => {
            closePopup();
            window.location.href = "dashboard.html"; // Redirect after login
        }, 2000);
    } else {
        loginMessage.style.display = "block";
        loginMessage.textContent = "Invalid credentials. Please try again.";
        loginMessage.style.color = "red";
    }
}

        function calculateBMI() {
            const height = document.getElementById('height').value;
            const weight = document.getElementById('weight').value;
            const bmiResult = document.getElementById('bmi-result');

            if (!height || !weight) {
                bmiResult.innerHTML = 'Please enter both height and weight.';
                return;
            }

            const heightInMeters = height / 100;
            const bmi = (weight / (heightInMeters ** 2)).toFixed(2);

            let category = '';
            if (bmi < 18.5) {
                category = 'Underweight';
            } else if (bmi >= 18.5 && bmi < 24.9) {
                category = 'Normal weight';
            } else if (bmi >= 25 && bmi < 29.9) {
                category = 'Overweight';
            } else {
                category = 'Obese';
            }

            bmiResult.innerHTML = `<strong>BMI:</strong> ${bmi} (${category})`;
        }

        function calculateCalories() {
        const age = document.getElementById('age').value;
        const height = document.getElementById('height').value;
        const weight = document.getElementById('weight').value;
        const gender = document.getElementById('gender').value;
        const activity = parseFloat(document.getElementById('activity').value);
        const result = document.getElementById('calorie-result');

        if (!age || !height || !weight) {
            result.innerHTML = 'Please fill in all fields.';
            return;
        }

        let bmr = gender === 'male' 
            ? 10 * weight + 6.25 * height - 5 * age + 5 
            : 10 * weight + 6.25 * height - 5 * age - 161;

        const calories = Math.round(bmr * activity);
        result.innerHTML = `<strong>Daily Calorie Needs:</strong> ${calories} kcal`;
    }

async function startChallenge() {
    const video = document.getElementById('webcam');
    const repDisplay = document.getElementById('rep-count');
    const exerciseDisplay = document.getElementById('current-exercise');
    const motivationText = document.getElementById('motivation');
    const boostImage = document.getElementById('boost-image');

    // Load PoseNet
    net = await posenet.load();

    // Access Webcam
    navigator.mediaDevices.getUserMedia({ video: true }).then((stream) => {
        video.srcObject = stream;
    });

    // Pose Detection Loop
    video.addEventListener('loadeddata', () => {
        detectPose(video);
    });

    async function detectPose(video) {
        const pose = await net.estimateSinglePose(video, { flipHorizontal: false });

        // Example: Count squats using knee position
        const leftKnee = pose.keypoints.find(p => p.part === 'leftKnee');
        const rightKnee = pose.keypoints.find(p => p.part === 'rightKnee');

        if (leftKnee && rightKnee && leftKnee.score > 0.7 && rightKnee.score > 0.7) {
            if (leftKnee.position.y > 400 && rightKnee.position.y > 400) {
                if (repCount < exercises[currentExercise].targetReps) {
                    repCount++;
                    repDisplay.textContent = repCount;
                }

                // Move to next exercise after reaching target
                if (repCount >= exercises[currentExercise].targetReps) {
                    motivationText.textContent = exercises[currentExercise].message;
                    boostImage.src = exercises[currentExercise].image;
                    boostImage.style.display = 'block';

                    currentExercise++;
                    if (currentExercise < exercises.length) {
                        setTimeout(() => {
                            repCount = 0;
                            exerciseDisplay.textContent = `Exercise: ${exercises[currentExercise].name}`;
                            repDisplay.textContent = repCount;
                            motivationText.textContent = '';
                            boostImage.style.display = 'none';
                        }, 2000);
                    } else {
                        // Final Congratulations
                        exerciseDisplay.textContent = "üéâ Challenge Completed! You're unstoppable!";
                        motivationText.textContent = "üèÜ Well Done, Champion!";
                        boostImage.src = "./Health/congrats.jpg"; // Replace with a real image
                    }
                }
            }
        }
        requestAnimationFrame(() => detectPose(video));
    }
}

    }
}

function playSound(key) {
    if (currentSound) {
        currentSound.pause();
        currentSound.currentTime = 0;
    }

    if (soundMap[key]) {
        currentSound = new Audio(soundMap[key]);
        currentSound.play();
    }
}

// Function to stop sound
function stopSound() {
    if (currentSound) {
        currentSound.pause();
        currentSound.currentTime = 0;
        currentSound = null;
    }
}

// Function to highlight key when pressed
function activateKey(key) {
    const keyElement = document.querySelector(`.key[data-key='${key}']`);
    if (keyElement) {
        keyElement.classList.add("active");
    }
}

// Function to remove key highlight when released
function deactivateKey(key) {
    const keyElement = document.querySelector(`.key[data-key='${key}']`);
    if (keyElement) {
        keyElement.classList.remove("active");
    }
}

      function trackCycle() {
    let lastPeriod = document.getElementById("last-period").value;
    let cycleLength = parseInt(document.getElementById("cycle-length").value);

    if (!lastPeriod || isNaN(cycleLength) || cycleLength < 21 || cycleLength > 35) {
        alert("Please enter valid data!");
        return;
    }

    let lastPeriodDate = new Date(lastPeriod);
    let nextPeriodDate = new Date(lastPeriodDate);
    nextPeriodDate.setDate(lastPeriodDate.getDate() + cycleLength);

    let today = new Date();

    document.getElementById("result").innerHTML = `<b>Next Period Date:</b> ${nextPeriodDate.toDateString()}`;

    // Pregnancy Detection (If missed period)
    if (today > nextPeriodDate) {
        let dueDate = new Date(lastPeriodDate);
        dueDate.setDate(lastPeriodDate.getDate() + 280); // 40 weeks from last period
